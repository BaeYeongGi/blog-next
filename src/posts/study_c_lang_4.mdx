## 31_포인터 배열_1

※ 배열의 요소가 포인터(메모리주소)로 이루어진 것

```c title="다음 C 프로그램의 실행 결과는?"
#include<stdio.h>
int main(){
  static char *c[] = {"aaa", "bbb", "ccc"};
  printf("%s", *(c+1));
}
// c 100 주소값이 들어가는 배열 
// 100번지 [20번지], 101번지 [30번지], 102번지 [40번지]
// 20번지 a a a
// 30번지 b b b
// 40번지 c c c

// *(c+1) → 101번지의 값 → 30번지의 값
// 답: bbb 

```

## 33_포인터 배열_3

```c title="다음 C 프로그램의 실행 결과는?"
void main(){
  int a[2][3] {{-3, 14, 5}, {1, -10, 8}}; // a[0], a[1] 에 해당
  int *b[] = {a[0], a[1]};
  int *p = b[1];
  printf("%d", *b[1]);
  printf("%d", *(++p));
  printf("%d", *(--p-2));
}

// int a 2개의 배열 공간 

// a 100번지
// 100 a[0] -3 14 5 
// 103 a[1] 1 -10 8
// *b[] 포인터 변수에는 a[0], a[1] → 100번지, 103번지 저장
// *p 포인터 변수에는 b[1] → a[1]에 해당하므로 103번지 저장

// *b[1] = 1 
// *(++p) → ++p → ++103번지 → 104번지의 값 → -10 
// *(--p-2) → 104 에서 -- → 103 - 2 → 101번지의 값 → 14
// 답: 1, -10, 14
```

## 36_배열포인터

* ※ 배열포인터?: 배열의 시작 주소값을 저장할 수 있는 포인터 2차원 배열의 구조. 그냥 하나의 포인터일 뿐

```c title="다음 C프로그램의 실행 결과로 옳은 것은?"
#include<stdio.h>
#include<stdlib.h>
#define N 3
int main(void){
  int (*in)[N], *out, sum = 0;
  in = (int (*)[N]) malloc(N*N*sizeof(int))
  out = (int*)in;
  for(int i = 0; i < N * N; i++) out[i] = i;
  for(int i = 0; i < N; i++)
    sum += in[i][i];
  printf("%d", sum);
  return 0;
}

// #define N 3 매크로: 앞으로 나오는 N은 3
// in 에 3 * 3 배열을 만듬
// in[0] 100번지 [] [] []
// in[1] 103번지 [] [] []
// in[2] 106번지 [] [] []

// 0 부터 N * N → 0 ~ 8까지 배열에 채워 넣음
// in[0] 100번지 [0] [1] [2]
// in[1] 103번지 [3] [4] [5]
// in[2] 106번지 [6] [7] [8]

// sum += in[i][i] 
// 반복문으로 0행0열, 1행1열, 2행2열

// sum = sum + in[i][i] 
// i = 1 / sum = sum 0 + 4
// i = 2 / sum = sum  4 + 8 
// i = 3 / 반목문 실행 x
// 답 sum = 12
```

## 37_함수

* ※ 함수란?: 반복적인 수행을 정의해 놓은 작은 프로그램 단위

```c title="함수 선언과 사용"
int sum(int a, int b)
{
  int c = a + b;
  return c;
}
int data = sum(10, 20);
printf("%d", data);

// return 30 → data = 30
// sum 함수 내에 선언됐던 변수들은 폐기됨
```

```c title="다음 C언어 프로그램의 실행 후 출력 결과로 옳은 것은?"
#include<stdio.h>
void swap(int a, int b){
  int temp;
  temp = a;
  a = b;
  b = temp;
}
int main(void){
  int k, j;
  k = 3;
  j = 2;
  swap(k, j);
  printf("k = %d, j = %d", k, j);
  return 0;
}

// swapping 알고리즘
// temp = 3, a = 2, b = 3
// swap 함수에는 return 값이 없고 아무일도 일어나지 않음 → 폐기
// 전역변수 or 정적변수라면 값을 보존
// k = 3, j = 2
```

## 38_함수와 반복문

```c title="다음 C언어 프로그램의 출력 값은?"
#include<stdio.h>
int func(int n);
int main(void){
  int num;
  printf("%d\n", func(5))
  return 0;
}

int func(int n){
  if(n < 2)
    return n;
  else {
    int i, tmp, curent = 1, last = 0;
    for(i = 2; i <= n; i++){
      tmp = curent;
      curent += last;
      last = tmp;
    }
    return curent;
  }
}

// func(int n) → 5
// int func(int 5)
// else 구문 실행
// i; tmp; curent = 1; last = 0;
// 반복문 내 실행
// i = 2 / tmp = 1; / curent = 1 + 0 / last = 1;
// i = 3 / tmp = 1; / curent = 1 + 1 / last = 1;
// i = 4 / tmp = 2; / curent = 2 + 1 / last = 2;
// i = 5 / tmp = 3; / curent = 3 + 2 / last = 3;
// i = 6 / 실행 x
// curent 5값을 func(5)로 호출
// 답: 5
```

## 39_함수에 주소전달_1

```c title="다음 C 언어 프로그램의 실행 결과는?"
#include<stdio.h>
int main(void){
  int x = 10, y = 20;
  printf("%d", f(&x, y));
  printf("%d %d\n", x, y);
}

int f(int *i, int j){
  *i += 5;
  return(2**i+ ++j);
}
// f(&x, y) → x의 주소값(100), y의 value
// int f(int *i, int j) → 100번지, 20 
// *i += 5 → 100번지를 참조하는 i의 값 → i = 10 + 5 
// x = 15
// return (2 * *i + ++j) → ++j먼저 수행 → 21
// 2 * i의 값 (15) → 30
// 30 + 21 = 51 을 return 
// f(&x, y) 는 51 
// x는 15, y는 20
// 답: 51, 15, 20
```

## 40_함수와 변수의 유효범위

```c title="다음 C 언어 프로그램의 실행 결과는?"
#include<stdio.h>
int a = 1, b = 2, c = 3;
int f(void);
int main(void){
  printf("%3d\n", f());
  printf("%3d%3d%3d\n", a, b, c);
  return 0;
}
int f(void){
  int b, c;
  a=b=c=4;
  return (a+b+c);
}

// %3d\n → 3개의 공간을 확보, f() 함수값값 대입
// a=b=c=4 → c = 4, b = 4, a는 함수내에 선언된 변수가 없고 전역으로 선언된 변수의 값 → 4
// (a + b + c) 4 + 4 + 4 = 12값 return
// printf("%3d\n", f()); → _ 1 2
// printf("%3d%3d%3d\n", a, b, c) → 4, 2, 3
// 답: _ 1 2, _ 4, _ 2, _ 3 

```
